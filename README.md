    """
    A function to balance work and fun.
    
    Args:
    - task (str): The task to execute.
    - mood (str): Your current mood. Default is 'neutral'.

    Returns:
    - str: A fun yet professional message.
    """
    fun_emojis = {
        'happy': 'üòÑ',
        'neutral': 'üòê',

        'sad': "Maybe take a short break and come back stronger."
